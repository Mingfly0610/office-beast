        let currentQuestionIndex = 0;

        let totalScore = 0;

        let currentResultTitle = "";


        const pageHome = document.getElementById('page-home');

        const pageQuiz = document.getElementById('page-quiz');

        const pageResult = document.getElementById('page-result');

        const progressBar = document.getElementById('progress-bar');

        const optionsContainer = document.getElementById('options-container');


        function switchPage(hidePage, showPage) {

            hidePage.classList.remove('active');

            showPage.classList.add('active');

        }


        function startQuiz() {

            currentQuestionIndex = 0;

            totalScore = 0;

            switchPage(pageHome, pageQuiz);

            renderQuestion();

        }


        function restartQuiz() {

            const activePage = document.querySelector('.page.active');

            switchPage(activePage, pageHome);

        }


        function renderQuestion() {

            const qData = questions[currentQuestionIndex];

            progressBar.style.width = `${(currentQuestionIndex / questions.length) * 100}%`;

            document.getElementById('question-counter').textContent = `${currentQuestionIndex + 1}/${questions.length}`;


            document.getElementById('quiz-card').style.opacity = 0;

            setTimeout(() => {

                document.getElementById('question-text').innerHTML = `Q${currentQuestionIndex + 1}. ${qData.q}`;

                optionsContainer.innerHTML = '';

                qData.options.forEach((opt) => {

                    const btn = document.createElement('button');

                    btn.className = "quiz-option w-full py-4 px-4 rounded-[20px] bg-white border-[3px] border-[#111] text-[#111] font-black text-lg text-center shadow-[4px_4px_0_#111] leading-snug";

                    btn.innerHTML = opt.text;

                    btn.onclick = () => handleOptionClick(btn, opt.score);

                    optionsContainer.appendChild(btn);

                });

                document.getElementById('quiz-card').style.opacity = 1;

            }, 200);

        }


        function handleOptionClick(btn, score) {

            btn.classList.add('selected');

            totalScore += score;

            setTimeout(() => {

                currentQuestionIndex++;

                if (currentQuestionIndex < questions.length) renderQuestion();

                else calculateAndShowResult();

            }, 500);

        }


        function calculateAndShowResult() {

            let resIdx = Math.min(Math.floor(totalScore / 2), 7);

            const res = resultsData[resIdx];

            currentResultTitle = res.title;

            document.getElementById('result-icon').textContent = res.icon;

            document.getElementById('result-title').textContent = res.title;

            document.getElementById('result-desc').textContent = res.desc;

            const tags = document.getElementById('result-tags');

            tags.innerHTML = '';

            res.tags.forEach(t => tags.innerHTML += `<span class="px-3 py-1 bg-white border-[2px] border-[#111] rounded-full text-xs font-black shadow-[2px_2px_0_#111]">#${t}</span>`);

            switchPage(pageQuiz, pageResult);

        }


        function showShareModal() {

            document.getElementById('share-preview-title').textContent = `我是【${currentResultTitle}】！`;

            document.getElementById('share-modal-overlay').classList.add('active');

            document.getElementById('share-modal').classList.add('active');

        }


        function closeShareModal() {

            document.getElementById('share-modal').classList.remove('active');

            document.getElementById('share-modal-overlay').classList.remove('active');

        }


        function copyResultLink() {

            const t = `我測出我是【${currentResultTitle}】！你也來測測看吧！\n${window.location.href}`;

            const i = document.createElement('input');

            document.body.appendChild(i); i.value = t; i.select(); document.execCommand('copy'); document.body.removeChild(i);

            const toast = document.getElementById('toast'); toast.textContent = "✅ 已複製連結！"; toast.classList.add('show');

            setTimeout(() => toast.classList.remove('show'), 2000);

            closeShareModal();

        }
